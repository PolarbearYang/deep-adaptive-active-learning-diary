[docs]class plotter(object):
    """A class used for plotting all information in an active round.

        Args:
            runner: (:obj:`query_strategies.strategy`)
                The strategy used in the task.
                The class :obj:`strategy` provides all information the plotter needs.
            dconfig: (dict)
                This should provide all data (train/eval/test) in raw data format:

                - 'train': {'X': X_train, 'Y': Y_train},
                - 'eval': {'X': X_eval, 'Y': Y_eval},
                - 'test': {'X': X_test, 'Y': Y_test}}

                New information will be updated in this dict when plotting.

    """
    def __init__(self, runner: strategy, dconfig: dict):
        self.runner = runner
        self.runner.args['logger'].info("Plotter Initializing...")
        self.dconfig = dconfig
        self.model = runner.net(num_classes=runner.args['num_classes']).to(runner.device)
        self.model.load_state_dict(self.runner.clf.state_dict())
        # self.outer_params = outer_params
        self._get_init_dinfo()
        self._split_predicted_data()
        self.save_path = os.path.join(self.runner.args['work_dir'],
                                      f'active_round_{runner.active_round}', 'figures')
        # 将范围划分为20个区域的函数
        self.get_bins = get_bins
        self.runner.args['logger'].info("Plotter Initialization Done!")

    def _get_init_dinfo(self):
        """Performed when initializing the class.
           Now it generates the following information (for train/eval/test mode):

           - pseudo: Model output of all samples in the last layer
           - entropy: The entropy of all samples corresponding to pseudo
           - features: The feature of all samples extracted by the model

        """
        for mode in ['train', 'eval', 'test']:
            loader = DataLoader(self.runner.handler(self.dconfig[mode]['X'], self.dconfig[mode]['Y'],
                                                    transform=self.runner.args['eval_transform']),
                                shuffle=False, batch_size=self.runner.args['batch_size'],
                                num_workers=self.runner.args['num_workers'])
            entropy_list = []
            pseudo_list = []
            # This step is for getting pseudo and entropy
            self.runner.args['logger'].info("Getting pseudo output and entropy...")
            with torch.no_grad():
                self.model.eval()
                for i, (x, y, _) in enumerate(track_iter_progress(loader)):  # 这里的data由(图像,标签)组成
                    x, y = x.to(self.runner.device), y.to(self.runner.device)
                    logits, _ = self.model(x, with_feature=True)
                    logits = logits.cpu()
                    softmax = torch.nn.Softmax(dim=1)
                    norm_output = softmax(logits)
                    pseudo_list.append(norm_output)
                    entropy_batch = torch.Tensor(scipy.stats.entropy(norm_output, axis=1))
                    entropy_list.append(entropy_batch)
            pseudo_out = torch.cat(pseudo_list, dim=0)  # 指的是模型产生的软标签，而非真实标签
            entropy_out = torch.cat(entropy_list, dim=0)  # 每个样本的entropy
            self.dconfig[mode]['pseudo'] = pseudo_out.numpy()
            self.dconfig[mode]['entropy'] = entropy_out.numpy()

            # This step is for getting features
            self.runner.args['logger'].info("Getting features...")
            feature_list = []
            with torch.no_grad():
                self.model.eval()
                for i, (x, y, _) in enumerate(track_iter_progress(loader)):  # 这里的data由(图像,标签)组成
                    x, y = x.to(self.runner.device), y.to(self.runner.device)
                    _, features = self.model(x, with_feature=True)
                    features = features.cpu()
                    feature_list.append(features)
            features_out = torch.cat(feature_list, dim=0)
            self.dconfig[mode]['features'] = features_out.numpy()

    def _split_predicted_data(self):
        """Performed when initializing the class.
           It will count the number of correctly labeled samples and the number of mislabeled samples in all 3 modes.
           Now it generates the following information (for train/eval/test mode):

           - entropy_list_correct: the entropy of all correctly labeled samples
           - entropy_list_incorrect: The entropy of all mislabeled samples
           - entropy_rank_list_correct: the ranking score corresponding to the entropy of all correctly labeled samples
           - entropy_rank_list_incorrect: the ranking score corresponding to the entropy of all mislabeled samples
           - idx_list_correct: the indices of all correctly labeled samples(corresponding to entropy_list_correct)
           - idx_list_incorrect: the indices of all mislabeled samples(corresponding to entropy_list_incorrect)

        """
        for mode in ['train', 'eval', 'test']:
            entropy_rank_list_correct = []
            entropy_rank_list_incorrect = []
            idx_list_correct = []
            idx_list_incorrect = []
            entropy_list_correct = []
            entropy_list_incorrect = []
            entropy_rank_out = get_rank_score(self.dconfig[mode]['entropy'])
            max_probs, max_idx = torch.max(torch.Tensor(self.dconfig[mode]['pseudo']), dim=-1)
            for i, gt_label in enumerate(self.dconfig[mode]['Y']):
                if max_idx[i] == gt_label:
                    entropy_rank_list_correct.append(entropy_rank_out[i])
                    entropy_list_correct.append(self.dconfig[mode]['entropy'][i])
                    idx_list_correct.append(i)
                else:
                    entropy_rank_list_incorrect.append(entropy_rank_out[i])
                    entropy_list_incorrect.append(self.dconfig[mode]['entropy'][i])
                    idx_list_incorrect.append(i)

            self.runner.args['logger'].info(f'Mislabeled cnt: {len(entropy_list_incorrect)} in {mode} set.')
            self.runner.args['logger'].info(f'Correctly labeled cnt: {len(entropy_list_correct)} in {mode} set.')

            entropy_list_correct = np.array(entropy_list_correct)
            entropy_list_incorrect = np.array(entropy_list_incorrect)

            self.dconfig[mode]['entropy_list_correct'] = entropy_list_correct
            self.dconfig[mode]['entropy_list_incorrect'] = entropy_list_incorrect
            self.dconfig[mode]['entropy_rank_list_correct'] = entropy_rank_list_correct
            self.dconfig[mode]['entropy_rank_list_incorrect'] = entropy_rank_list_incorrect
            self.dconfig[mode]['idx_list_correct'] = idx_list_correct
            self.dconfig[mode]['idx_list_incorrect'] = idx_list_incorrect


[docs]    def bar_by_class(self, mode='train'):
        """Plot the all kinds of bar plot, which includes:

        - The number of samples corresponding to different entropy levels.
        - The number of samples corresponding to different entropy levels(by rank).
        - The number of correctly labeled samples and mislabeled samples by the model in each class.
        - The number of mislabeled samples by the model in each class.
        - The number of labeled samples in the current training pool in each class.
        - The number of queried samples in the current round in each class.

        :param mode: (str) Which data split we are using. Choose one from `train`, `eval`, `test`.

        - In the arg `mode`:
            - `train` means we are using the `train` split dataset.
                - Information related to query/label&unlabel are only available in this part.
            - `eval`/`test` means we are using `eval`/`test` split in the dataset.

        """
        save_path = os.path.join(self.save_path, mode)
        os.makedirs(save_path, mode=0o777, exist_ok=True)
        # 标签数字到对应类名称的映射
        label2text = self.runner.args['label2text']
        # 逐类统计
        correct_by_class = np.zeros(len(label2text), dtype=int)
        incorrect_by_class = np.zeros(len(label2text), dtype=int)
        text_by_class = list(label2text.values())

        # 查看错误样本/正确样本的标注
        Y_correct_temp = self.dconfig[mode]['Y'][self.dconfig[mode]['idx_list_correct']]
        Y_incorrect_temp = self.dconfig[mode]['Y'][self.dconfig[mode]['idx_list_incorrect']]
        for i, label in enumerate(label2text.keys()):
            correct_by_class[i] = torch.sum(Y_correct_temp == label).item()
            incorrect_by_class[i] = torch.sum(Y_incorrect_temp == label).item()

        bar_width = 0.3  # 条形宽度
        index_correct = np.arange(len(text_by_class))  # 正确样本条形图的横坐标
        index_incorrect = index_correct + bar_width  # 错误条形图的横坐标

        # Number of samples corresponding to different entropy levels
        bar_by_entropy(save_path, self.dconfig[mode]['entropy_list_correct'],
                       self.dconfig[mode]['entropy_list_incorrect'], self.get_bins)
        bar_by_rank_entropy(save_path, self.dconfig[mode]['entropy_rank_list_correct'],
                            self.dconfig[mode]['entropy_rank_list_incorrect'], self.get_bins)

        # CORRECT AND INCORRECT NUM BY CLASS TOGETHER
        bar_correct_incorrect_by_class(save_path, index_correct, index_incorrect,
                                       correct_by_class, incorrect_by_class,
                                       bar_width, text_by_class)
        # INCORRECT NUM BY CLASS
        bar_incorrect_by_class(save_path, text_by_class, incorrect_by_class)

        # 查看标注样本的标注
        if mode == 'train':
            # LABELED NUM BY CLASS
            bar_labeled_by_class(save_path, label2text, text_by_class,
                                 self.dconfig[mode]['Y'], self.runner.idxs_lb)
            # QUERIED NUM BY CLASS
            queried_by_class = np.zeros(len(label2text), dtype=int)
            queried_correct_by_class = np.zeros(len(label2text), dtype=int)
            queried_incorrect_by_class = np.zeros(len(label2text), dtype=int)
            queried_temp = self.dconfig[mode]['Y'][self.runner.idxs_q]
            queried_correct_temp = self.dconfig[mode]['Y'][
                list(set(self.runner.idxs_q).intersection(set(self.dconfig[mode]['idx_list_correct'])))]
            queried_incorrect_temp = self.dconfig[mode]['Y'][
                list(set(self.runner.idxs_q).intersection(set(self.dconfig[mode]['idx_list_incorrect'])))]
            for i, label in enumerate(label2text.keys()):
                queried_by_class[i] = torch.sum(queried_temp == label).item()
                queried_correct_by_class[i] = torch.sum(queried_correct_temp == label).item()
                queried_incorrect_by_class[i] = torch.sum(queried_incorrect_temp == label).item()
            bar_queried_by_class(save_path, queried_by_class, text_by_class)
            bar_queried_correct_incorrect_by_class(save_path, index_correct, index_incorrect,
                                                   queried_correct_by_class, queried_incorrect_by_class,
                                                   text_by_class, bar_width)
            bar_queried_incorrect_by_class(save_path, queried_incorrect_by_class, text_by_class)



[docs]    def classification_with_cluster(self, method='tsne', mode='train', use_predefined=False):
        """Plot the all kinds of figures with features embedded into a 2D space:

        - The current distribution of correctly classified and misclassified samples in 2D space
        - The current distribution of samples with sample-wise specified values: e.g. entropy, acquisition score, weight
        - The current labeled and unlabeled samples in 2D space
        - The current distribution of queried samples in 2D space
        - The current distribution embedding in 2D space

        :param method: (str) The dimension reduction method we choose. Choose one from `raw`, `tsne`, `umap`.
        :param mode: (str) Which data split we are using. Choose one from `train`, `eval`, `test`.
        :param use_predefined: (bool) Whether to use predefined 2D results to represent the distribution. This is only available in `train` mode.

        - In the arg `method`:
            - `raw` means the given data are already two-dimensional, no reduction method will be applied
            - `tsne` means the data will be reduced to 2-D space using :obj:`sklearn.manifold.TSNE`
            - `umap` means the data will be reduced to 2-D space using :obj:`umap.UMAP`
        - In the arg `mode`:
            - `train` means we are using the `train` split dataset.
                - Cluster methods related to query/label&unlabel are only available in this part.
            - `eval`/`test` means we are using `eval`/`test` split in the dataset.

        """
        # 这个尽量在使用默认训练的模型时使用，更容易显示出错误分布的位置
        assert method in ['raw', 'tsne', 'umap']
        if use_predefined:
            save_path = os.path.join(self.save_path, mode, 'with_predefined')
        else:
            save_path = os.path.join(self.save_path, mode)
        os.makedirs(save_path, mode=0o777, exist_ok=True)
        if method == 'raw':
            X_embedded = self.dconfig[mode]['X']
        if method == 'tsne':
            if self.runner.args['plot_using_predefined'] and mode == 'train' and use_predefined:
                if os.path.exists(self.runner.args['plot_preload_path']):
                    X_embedded = np.load(self.runner.args['plot_preload_path'])
                    self.runner.args['logger'].info(f"TSNE Embedding loaded from "
                                                    f"{self.runner.args['plot_preload_path']}")
                else:
                    self.runner.args['logger'].info("preload path does not exist! Skip plotting!")
                    return
            elif os.path.exists(os.path.join(self.save_path, f'tsne_embedded_{mode}.npy')):
                X_embedded = np.load(os.path.join(self.save_path, f'tsne_embedded_{mode}.npy'))
                self.runner.args['logger'].info(f"TSNE Embedding loaded from "
                                                f"{os.path.join(self.save_path, f'tsne_embedded_{mode}.npy')}")
            else:
                X_embedded = TSNE(n_components=2, random_state=2021)\
                    .fit_transform(self.dconfig[mode]['features'])
                np.save(os.path.join(self.save_path, f'tsne_embedded_{mode}.npy'), X_embedded)
                self.runner.args['logger'].info(f"TSNE Embedding saved to "
                                                f"{os.path.join(self.save_path, f'tsne_embedded_{mode}.npy')}!")
        if method == 'umap':
            if self.runner.args['plot_using_predefined'] and mode == 'train' and use_predefined:
                if os.path.exists(self.runner.args['plot_preload_path']):
                    X_embedded = np.load(self.runner.args['plot_preload_path'])
                    self.runner.args['logger'].info(f"UMAP Embedding loaded from "
                                                    f"{self.runner.args['plot_preload_path']}")
                else:
                    self.runner.args['logger'].info("preload path does not exist! Skip plotting!")
                    return
            elif os.path.exists(os.path.join(self.save_path, f'umap_embedded_{mode}.npy')):
                X_embedded = np.load(os.path.join(self.save_path, f'umap_embedded_{mode}.npy'))
                self.runner.args['logger'].info(f"UMAP Embedding loaded from "
                                                f"{os.path.join(self.save_path, f'umap_embedded_{mode}.npy')}")
            else:
                X_embedded = umap.UMAP(random_state=2021)\
                    .fit_transform(self.dconfig[mode]['features'])
                np.save(os.path.join(self.save_path, f'umap_embedded_{mode}.npy'), X_embedded)
                self.runner.args['logger'].info(f"UMAP Embedding saved to "
                                                f"{os.path.join(self.save_path, f'umap_embedded_{mode}.npy')}!")

        # 按照类别进行绘画，同一个类别使用同一种颜色
        style = ['o'] * self.dconfig[mode]['Y'].shape[0]
        for idx in self.dconfig[mode]['idx_list_incorrect']:
            style[idx] = 'x'
        style = np.array(style)

        indices = self.dconfig[mode]['idx_list_incorrect']
        indices = np.hstack([self.dconfig[mode]['idx_list_correct'], indices])

        sns.set(rc={'figure.figsize': (11.7, 8.27)})

        # Plot the distribution obtained by the current model with embedding
        cluster_with_embedding(save_path, self.runner.args['num_classes'], X_embedded,
                               np.array([self.runner.args['label2text'][x.item()]
                                         for x in self.dconfig[mode]['Y'][indices]]),
                               indices, style, method='tsne')

        # 按照对错进行绘画（主要用于预训练参数拟合得很好的时候）
        cluster_with_correct_or_not(save_path, X_embedded, self.dconfig[mode]['Y'],
                                    self.dconfig[mode]['idx_list_correct'],
                                    indices, method='tsne')

        # 按照ENTROPY绘画，ENTROPY越大颜色越深
        # 这个entropy超级慢! 是for循环的问题!
        # 本来是想看每个样本的entropy/得分/软加权等信息，现在这一步暂且跳过，有待优化
        # cluster_with_value(save_path, 'entropy', self.dconfig[mode]['entropy'],
        #                    X_embedded, indices, method='tsne')

        # 再按照是否标注进行绘画（主要用来看标注样本相对于整个集合的分布）
        if mode == 'train':
            cluster_with_lab_unlab(save_path, X_embedded, self.runner.idxs_lb, method)
            cluster_with_query(save_path, self.runner.args['num_classes'], X_embedded, self.runner.idxs_q,
                               np.array([self.runner.args['label2text'][x.item()]
                                         for x in self.dconfig[mode]['Y'][indices]]), indices, method='tsne')



[docs]    def check_inconsistency(self, plotter_B):
        """Compare the information with the last active round(specified by another plotter).

            :param plotter_B: Another plotter of the same class :obj:`plotter.plotter`

        """

        joint_union_info = {}
        for mode in ['train', 'eval', 'test']:
            joint_union_info[mode] = {}
            incorrect_data_union = set(self.dconfig[mode]['idx_list_incorrect'])\
                .union(set(plotter_B.dconfig[mode]['idx_list_incorrect']))
            incorrect_data_joint = set(self.dconfig[mode]['idx_list_incorrect'])\
                .intersection(set(plotter_B.dconfig[mode]['idx_list_incorrect']))

            idx_list_incorrect = self.dconfig[mode]['idx_list_incorrect']
            plotter_b_idx_list_incorrect = plotter_B.dconfig[mode]['idx_list_incorrect']
            self.runner.args['logger'].info(
                f'Inconsistency Info in {mode} part:\n'
                f'Number of Mislabeled data in the last active round: {len(set(idx_list_incorrect))}\n'
                f'Number of Mislabeled data in the current active round: {len(set(plotter_b_idx_list_incorrect))}\n'
                f'Union: {len(incorrect_data_union)}\n'
                f'Intersection: {len(incorrect_data_joint)}\n')

            joint_union_info[mode]['incorrect_data_joint'] = incorrect_data_joint
            joint_union_info[mode]['incorrect_data_union'] = incorrect_data_union

        return joint_union_info
