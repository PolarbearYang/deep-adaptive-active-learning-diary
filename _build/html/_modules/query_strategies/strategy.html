class Strategy:
    """The base class wrapping the training and querying procedure.
    All new strategies should be inherited from this class.

    Typically in a new strategy, we can just overload the `query` member function,
    which returns the indices of queried samples.

    Args:
        X: (Any)
            Any form of raw data. It should be the `train` split of the dataset.
        Y: (Any)
            The labels of X.
        idxs_lb: np.ndarray(bool)
            An boolean array that has the same length with the `train` split of the dataset.
        net: (class)
            A net wrapper, with a `__call__` member that returns the model architecture,
            its parameters are just the parameters of the model class,
            arg `num_classes` is necessary.
        handler: (torch.utils.data.Dataset)
            Dataset wrappers designed for active learning, whose `__getitem__` returns (data, label, idx)
        args: (dict)
            Contains lots of information the strategy needs for training,
            including the ones in the parser and some newly added ones.
            Here we list some of them for detailed explanation.

            - `work_dir`: (str) The working directory, in which we save checkpoints, log info, figures
            - `save_freq`: (int) Save a checkpoint per `save_freq` epochs.
            - `n_epoch`: (int) How many epochs are in one active round.
            - `batch_size`: (int) Just the batch size of the dataloader.
            - `num_workers`: (int) `num_workers` of the dataloader.
            - `lr`: (float) Learning rate of the optimizer.
            - `momentum`: (float) Momentum of the optimizer.
            - `out_iter_freq`: (int) Print a log message per `out_iter_freq` iters.
            - `num_classes`: (int) The number of classes in the dataset.
            - `label2text`: (dict[int, str]) The class name corresponding to each label.
            - `train_transform`: (:obj:`torchvision.transforms.Compose`)Transformation applied to the training data.
            - `eval_transform`: (:obj:`torchvision.transforms.Compose`)Transformation applied to the eval/test data.
            - `timestamp`: (str) The start timestamp of the process.
            - `logger`: (:obj:`logging.RootLogger`) The logger to use.

    """
    def __init__(self, X, Y, idxs_lb, net, handler, args):
        self.X = X
        self.Y = Y
        self.idxs_lb = idxs_lb
        self.idxs_q = []
        self.handler = handler
        self.args = args
        self.net = net  # This is a function that returns a model structure, not the model itself!
        self.n_pool = len(Y)
        use_cuda = torch.cuda.is_available()
        self.device = torch.device("cuda" if use_cuda else "cpu")
        self.clf = net(num_classes=self.args['num_classes']).to(self.device)
        self.clf.apply(weight_init)
        self.embed_dimension = 50  # The dimension of space to perform distance based strategies
        self.active_round = 0
        self.logger = args['logger']
        self.TextLogger = TextLogger(self.clf, args)
        self.optimizer = optim.SGD(self.clf.parameters(), lr=self.args['lr'], momentum=self.args['momentum'])
        self.timer = Timer()
        self.epoch = 0   # This is for resume
        self.acc_list = None
        self.num_labels_list = None

    def _train(self, loader_tr):
        """Represents one epoch.

        :param loader_tr: (:obj:`torch.utils.data.DataLoader`) The training data wrapped in DataLoader.

        Accuracy and loss in the each iter will be recorded.

        """
        iter_out = self.args['out_iter_freq']
        loss_list = []
        right_count_list = []
        samples_per_batch = []
        self.clf.train()
        for batch_idx, (x, y, idxs) in enumerate(loader_tr):
            x, y = x.to(self.device), y.to(self.device)
            self.optimizer.zero_grad()
            out, e1 = self.clf(x)
            pred = out.max(1)[1]
            right_count_list.append((pred == y).sum().item())
            samples_per_batch.append(len(y))
            loss = F.cross_entropy(out, y)
            loss_list.append(loss.item())
            loss.backward()
            self.optimizer.step()
            iter_time = self.timer.since_last_check()
            if (batch_idx + 1) % iter_out == 0:
                log_dict = dict(
                    mode='train',   # 训练模式
                    epoch=self.epoch,   # 当前是第几个epoch
                    iter=batch_idx + 1,   # 当前是第几个iter
                    lr=get_lr(self.optimizer),   # 获取当前optimizer的学习率
                    time=iter_time,   # 当前iter消耗的时间
                    acc=1.0 * np.sum(right_count_list[-iter_out:]) / np.sum(samples_per_batch[-iter_out:]),
                    loss=np.sum(loss_list[-iter_out:])
                )
                self.TextLogger.log(
                    log_dict=log_dict,
                    iters_per_epoch=len(loader_tr),
                    iter_count=self.epoch * len(loader_tr) + batch_idx,
                    max_iters=self.args["n_epoch"] * len(loader_tr),  # 一个active round总共的epoch数量
                    interval=iter_out   # 多少个iter进行一次log
                )


[docs]    def train(self, resumed=False):
        """Represents one epoch.

            :param resumed: (bool) A flag indicating whether to resume from an uncomplete task.

        """
        self.logger.info('Start running, host: %s, work_dir: %s',
                         f'{getuser()}@{gethostname()}', self.args['work_dir'])
        self.logger.info('max: %d epochs', self.args['n_epoch'])
        self.clf.train()
        n_epoch = self.args['n_epoch']
        if not resumed:
            self.epoch = 0
            self.optimizer = optim.SGD(self.clf.parameters(), lr=self.args['lr'], momentum=self.args['momentum'])
        idxs_train = np.arange(self.n_pool)[self.idxs_lb]
        loader_tr = DataLoader(self.handler(self.X[idxs_train], self.Y[idxs_train],
                                            transform=self.args['train_transform']),
                               shuffle=True, batch_size=self.args['batch_size'],
                               num_workers=self.args['num_workers'])

        while self.epoch < n_epoch:
            self.timer.since_last_check()
            self._train(loader_tr)
            if self.epoch % self.args['save_freq'] == 0 and self.epoch > 0:
                self.save()
            self.save_for_resume()
            self.epoch += 1
        self.save()



[docs]    def predict(self, X, Y):
        """To predict the accuracy of some data split.
        We typically input the raw data of eval/test split here.

            :param X: (Any) Any sort of raw data.
            :param Y: (Any) Labels corresponding to X.

            Returns:
                torch.Tensor: The output label corresponding to each sample.

        """
        loader_te = DataLoader(self.handler(X, Y, transform=self.args['eval_transform']),
                               shuffle=False, batch_size=self.args['batch_size'],
                               num_workers=self.args['num_workers'])

        self.clf.eval()
        P = torch.zeros(len(Y), dtype=Y.dtype)
        with torch.no_grad():
            for x, y, idxs in track_iter_progress(loader_te):
                x, y = x.to(self.device), y.to(self.device)
                out, e1 = self.clf(x)

                pred = out.max(1)[1]
                P[idxs] = pred.cpu()

        return P



[docs]    def predict_eval(self, X, Y, mode='val', rd=0):
        """To get the predicted accuracy of the current data split.
        We typically input the raw data of eval/test split here.

            :param X: (Any) Any sort of raw data.
            :param Y: (Any) Labels corresponding to X.
            :param mode: (str) Choose `val` or `test`, in mode `val`, the number of active round will be recorded.
            :param rd: (int) The current active round.

            Returns:
                float: The output label corresponding to each sample.

        """
        P = self.predict(X, Y)
        acc = 1.0 * (Y == P).sum().item() / len(Y)
        if mode == 'val':
            log_dict = dict(
                mode=mode,
                active_round=rd,
                acc=acc
            )
        else:
            log_dict = dict(
                mode=mode,
                acc=acc
            )
        self.TextLogger.log(log_dict)
        return acc

        """To get the output probability distribution of the current data split.

            :param X: (Any) Any sort of raw data.
            :param Y: (Any) Labels corresponding to X.

            Returns:
                torch.Tensor: output probability distribution of all samples in the current data split

        loader_te = DataLoader(self.handler(X, Y, transform=self.args['eval_transform']),

                               num_workers=self.args['num_workers'])

        self.clf.eval()
        probs = torch.zeros([len(Y), len(np.unique(Y))])
        with torch.no_grad():
            for x, y, idxs in track_iter_progress(loader_te):
                x, y = x.to(self.device), y.to(self.device)
                out, e1 = self.clf(x)
                prob = F.softmax(out, dim=1)
                probs[idxs] = prob.cpu()

[docs]    def predict_prob_dropout(self, X, Y, n_drop):
        """To get the output probability distribution of the current data split.
        Output the average of all dropout versions.

            :param X: (Any) Any sort of raw data.
            :param Y: (Any) Labels corresponding to X.
            :param n_drop: (int) The number of times to perform dropout(Only available when there is :obj:`torch.nn.Dropout` module in the model).

            Returns:
                torch.Tensor: output probability distribution of all samples in the current data split.
                Actually we output the average distribution of all dropout versions.

        """
        loader_te = DataLoader(self.handler(X, Y, transform=self.args['eval_transform']),
                               shuffle=False, batch_size=self.args['batch_size'],
                               num_workers=self.args['num_workers'])

        self.clf.train()
        probs = torch.zeros([len(Y), len(np.unique(Y))])
        for i in range(n_drop):
            self.logger.info('n_drop {}/{}'.format(i+1, n_drop))
            with torch.no_grad():
                for x, y, idxs in track_iter_progress(loader_te):
                    x, y = x.to(self.device), y.to(self.device)
                    out, e1 = self.clf(x)
                    prob = F.softmax(out, dim=1)
                    probs[idxs] += prob.cpu()
        probs /= n_drop
        
        return probs

        """To get the feature embedding of the current data split.

            :param X: (Any) Any sort of raw data.
            :param Y: (Any) Labels corresponding to X.

            Returns:
                torch.Tensor: The feature embedding of all samples in the current data split.

        with torch.no_grad():
            for x, y, idxs in track_iter_progress(loader_te):
                x, y = x.to(self.device), y.to(self.device)
                out, e1 = self.clf(x)
                embedding_list.append(e1)
        embedding = torch.cat(embedding_list, dim=0)
        
        return embedding



[docs]    def save(self):
        """Save the current model parameters."""
        model_out_path = Path(os.path.join(self.args['work_dir'], f'active_round_{self.active_round}'))
        state = self.clf.state_dict(),
        if not model_out_path.exists():
            model_out_path.mkdir()
        save_target = model_out_path / f"active_round_{self.active_round}-" \
                                       f"label_num_{np.sum(self.idxs_lb).item()}-epoch_{self.epoch}.pth"
        torch.save(state, save_target)

        self.logger.info('==> save model to {}'.format(save_target))



[docs]    def save_for_resume(self):
        """Save the all the information for resuming
        (Deal with the case when the procedure unexpectly terminates.)"""
        model_out_path = Path(self.args['work_dir'])
        state = {
            "active_round": self.active_round,
            "epoch": self.epoch,
            "optimizer_param_groups": self.optimizer.param_groups,
            "idxs_lb": self.idxs_lb,
            "weight": self.clf.state_dict(),
            "acc": self.acc_list,
            "num_labels": self.num_labels_list
        }
        save_target = model_out_path / f"latest.pth"
        torch.save(state, save_target)
        self.logger.info(f'Saving Checkpoint at active round {self.active_round}'
                         f', epoch {self.epoch}')



[docs]    def resume(self):
        """Resume from the last checkpoint."""
        model_save_path = Path(self.args['work_dir'])
        load_target = model_save_path / f"latest.pth"
        if os.path.exists(load_target):
            state = torch.load(load_target)
            self.active_round = state["active_round"]
            self.epoch = state["epoch"]
            self.idxs_lb = state["idxs_lb"]
            self.clf.load_state_dict(state["weight"])
            self.optimizer.param_groups = state["optimizer_param_groups"]
            self.TextLogger.start_iter = self.epoch * (np.sum(self.idxs_lb).item() // self.args['batch_size'] + 1)
            self.logger.info(f'Resume from checkpoint at active round {self.active_round}'
                             f', epoch {self.epoch}')
            return self.active_round, state["acc"], state["num_labels"]
        else:
            print("Training from the beginning!")
            return None
