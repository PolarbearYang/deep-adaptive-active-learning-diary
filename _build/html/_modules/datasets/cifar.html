[docs]class CIFARHandler(Dataset):
    """A dataset wrapping the samples of CIFAR10 and CIFAR100

        Args:
            X: np.array
                An array consists of 32-by-32 images.
                Raw data should be presented here.
            Y: list
                A list consists of all labels corresponding to images in X.
            transform:
                Any form of transformations.

    """
    def __init__(self, X, Y, transform=None):
        self.X = X
        self.Y = Y
        self.transform = transform

    def __getitem__(self, index):
        x, y = self.X[index], self.Y[index]
        if self.transform is not None:
            x = Image.fromarray(x)
            x = self.transform(x)
        return x, y, index

    def __len__(self):
        return len(self.X)

class cifar10(object):
    """A class used for obtaining CIFAR10 raw data.
       This should be regarded as a function.

       Returns:
           X_train: np.ndarray
               The Raw data of the training set.
               Splitted from the original training set (54000 images).
           Y_train: list
               The labels of the X_train.
           X_eval: np.ndarray
               The Raw data of the validation set.
               Splitted from the original training set (6000 images).
           Y_eval: list
               The labels of the X_eval.
           X_test: np.ndarray
               The Raw data of the test set.
               Consists of 10000 images.
           Y_test: list
               The labels of the X_test.
           num_classes: int
               The number of classes is 10.
           idx_to_label: dict[int, str]
               The class name corresponding to each label.
           handler: :obj:`CIFARHandler`
               The dataset wrapper of the current dataset
           wrapper: :obj:`cifar10_wrapper`
               A wrapper that can return the transformations(train/eval) of the current dataset

        """
    def __init__(self, split_ratio=0.9):
        data_tr = datasets.CIFAR10('/home/yangjianan/deep-active-learning/data/cifar10', train=True, download=True)
        data_te = datasets.CIFAR10('/home/yangjianan/deep-active-learning/data/cifar10', train=False, download=True)
        data_length = len(data_tr.data)
        self.X_train = data_tr.data[:int(data_length * split_ratio)]
        self.Y_train = torch.from_numpy(np.array(data_tr.targets[:int(data_length * split_ratio)]))
        self.X_eval = data_tr.data[int(data_length * split_ratio):]
        self.Y_eval = torch.from_numpy(np.array(data_tr.targets[int(data_length * split_ratio):]))
        self.X_test = data_te.data
        self.Y_test = torch.from_numpy(np.array(data_te.targets))
        self.handler = CIFARHandler
        self.wrapper = cifar10_wrapper

    def __call__(self):
        return self.X_train, self.Y_train,\
               self.X_eval, self.Y_eval,\
               self.X_test, self.Y_test,\
               10, temp_label_dict['cifar10'], self.handler, self.wrapper()
